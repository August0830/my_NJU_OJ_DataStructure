```
struct Node
{
public:
	int data;
	int index;
	Node* next;
	Node(int val, int ptr, Node*pt=NULL)
	{
		data = val;
		index = ptr;
		next = pt;
	}
};
struct LinkNode
{
public:
	Node* top;
	int n;
	LinkNode(int cnt = 0) 
	{ top = NULL; n = cnt; };
	bool push(Node newnode);
	bool pop();
	void Print();
};

LinkNode stack;
	stack.n = 5;
	for (int i = 0; i < 5; ++i)
	{
		Node tmp(i, i + 100);
		stack.push(tmp);
	}
	stack.Print(); cout << endl;
	for (int i = 0; i < 5; ++i)
	{
		stack.pop();
		stack.Print(); cout << endl;
	}
	return 0;
}
bool LinkNode::push(Node newnode)
{
	if (top == NULL)
		top = &newnode;
	else
	{
		newnode.next = top;
		top = &newnode;
	}
	return true;
}
bool LinkNode::pop()
{
	if (top == NULL)
		return false;
	Node* tmp = top;
	top = top->next;
	delete tmp;
	return true;
}
void LinkNode::Print()
{
	Node* tmp = top;
	for (int i = 0; i < n; ++i)
	{
		cout << tmp->data << " " << tmp->index;
		tmp = tmp->next;
	}
}
```
